        QuadraticFild res(1,0,w,p), a = *this;

struct Two
{
    ll n,p;
    Two() {}
    Two(ll n,ll p):n(n),p(p) { }
    
    ll fastpow(ll a,ll b,ll m)
    {
        ll res = 1;
        while(b) {
            if(b&1) res = res * a % m;
            b >>= 1;
            a = a * a % m;
        }
        return res;
    }

    //æ±‚å‹’è®©å¾·ç¬¦å·
    ll Legender(ll a,ll p) {
        ll ans = fastpow(a,(p-1)>>1,p);
        if(ans + 1 == p) return -1;

    }
