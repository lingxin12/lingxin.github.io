#include<bits/stdc++.h>
#define enl puts("")
#define pii pair<int,int>
#define pb push_back
typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
using namespace std;
const int maxn = 1e5 + 10;
const int maxm = 4e5 + 10;
const int inf = 0x3f3f3f3f;

int d[maxn],fir[maxn], color[maxm] , to[maxm] , nxt[maxm], resk, rescolor[maxn],vis[maxn];
int n,m,a,b,c,edgei;
queue<int > q;

inline void intit()
{
	memset(fir,-1,sizeof(fir));
	memset(vis,0,sizeof(vis));
	edgei = 0;
	while(!q.empty()) q.pop();
}

inline void addedge(int a,int b,int c)
{
	nxt[++edgei] = fir[a];
	to[edgei] = b; color[edgei] = c;
	fir[a] = edgei;
}

inline void solve()
{
	while(!q.empty())
	{
		int from = q.front(); q.pop();
		for(int i=fir[from];i!=-1;i=nxt[i]) {
			if(d[to[i]] > d[from]+1) {
				d[to[i]] = d[from]+1;
				q.push(to[i]);
			}
		}
	}
	resk = d[1];
}

inline void bfs()
{
	for(int kk=1;kk<=resk;kk++)
	{
		int qlen = q.size(), mincolor = inf;
		for(int k=1;k<=qlen;k++) {
			int from = q.front(); q.push(q.front()); q.pop();
			for(int i=fir[from];i!=-1;i=nxt[i]) if(d[to[i]]==d[from]-1) mincolor = min(mincolor, color[i]);
		}
		rescolor[kk] = mincolor;
		if(kk == resk) return;
		for(int k=1;k<=qlen;k++) {
			int from = q.front(); q.pop();
			for(int i=fir[from];i!=-1;i=nxt[i]) 
				if(color[i] == mincolor && d[to[i]]==d[from]-1 && !vis[to[i]])
					q.push(to[i]), vis[to[i]] = 1;
		}
	}
}

int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		intit();
		for(int i=1;i<=m;i++) {
			scanf("%d%d%d",&a,&b,&c);
			if(a == b) continue;
			addedge(a,b,c);
			addedge(b,a,c);
		}
		for(int i=1;i<=n;i++) d[i] = inf;
		d[n] = 0;
		q.push(n);
		solve();
		q.push(1);
		bfs();
		printf("%d\n",resk);
		for(int i=1;i<=resk;i++) printf("%d%c",rescolor[i],i==resk?'\n':' ');
	}
	return 0;
}
